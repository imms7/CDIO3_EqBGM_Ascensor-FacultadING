
#include <Servo.h>
#include <LiquidCrystal.h>
Servo servo1;
LiquidCrystal lcd(4, 5, 6, 7, 8, 9); //rs, en, d4, d5, d6, d7
int pul_0 = 10;
int pul_1 = 11;
int pul_2 = 12;
int pul[] = {10, 11, 12};
int vpul[] = {1, 1, 1}; // ninguno seleccionado
int actual = 1;
void setup() {
  pinMode(pul[0], INPUT);
  pinMode(pul[1], INPUT);
  pinMode(pul[2], INPUT);
  pinMode(13, OUTPUT);
  pinMode(2, OUTPUT);
  pinMode(A3, OUTPUT);
  pinMode(A0, OUTPUT);
  pinMode(A1, OUTPUT);
  pinMode(A2, OUTPUT);
  servo1.attach(2);
  lcd.begin(16, 2);
  lcd.clear();
  lcd.setCursor(0, 0);
  lcd.print("No. de piso: ");
  delay(1000);
  Serial.begin(9600);
  servo1.write(93);
  lcd.setCursor(13, 0);
  lcd.print("P1");
  analogWrite(A0,255);
  // 0 Y 180 MAX VELOCIDAD EN AMBOS SENTIDOS
  // servo.write(90); DETENER. ENTRE 0 Y 90 SE VA FRENANDO
  // servo.write(180); ANTIHORARIO. ENTRE 90 Y 180 VA ACELERANDO
  // servo.write(0); HORARIO
}
void loop() {
  vpul[0] = digitalRead(pul[0]);
  vpul[1] = digitalRead(pul[1]);
  vpul[2] = digitalRead(pul[2]);
  digitalWrite(13, LOW);
  if (vpul[0] == LOW) { //ir a piso P1
    digitalWrite(13, HIGH);
    lcd.setCursor(13, 0);
    lcd.print("P1");
    switch (actual) {
      case 2:
        servo1.write(0);
        lcd.setCursor(0, 1);
        lcd.print("Bajando");
        delay(1000);
        servo1.write(93);
        lcd.print("       ");
        actual = 1;
        analogWrite(A0,255);
        analogWrite(A1,0);
        analogWrite(A2,0);
        break;
      case 3:
        servo1.write(0);
        lcd.setCursor(0, 1);
        lcd.print("Bajando");
        delay(2000);
        servo1.write(93);
        lcd.print("       ");
        actual = 1;
        analogWrite(A0,255);
        analogWrite(A1,0);
        analogWrite(A2,0);        
      break;
      case 1:
        lcd.setCursor(0, 1);
        lcd.print("Ya estas en P1");
        delay(1000);
        servo1.write(93);
        lcd.print("       ");
        actual = 1;
        analogWrite(A0,255);
        analogWrite(A1,0);
        analogWrite(A2,0);
        break;
    }
    delay(1000);
  }
  else if (vpul[1] == LOW) { // Ir a piso 2
    digitalWrite(13, HIGH);
    lcd.setCursor(13, 0);
    lcd.print("02");
    switch (actual) {
      case 1:
        servo1.write(180);
        lcd.setCursor(0, 1);
        lcd.print("Subiendo");
        delay(1000);
        servo1.write(93);
        lcd.print("       ");
        actual = 2;
        analogWrite(A0,0);
        analogWrite(A1,255);
        analogWrite(A2,0);
      break;
      case 2:
        lcd.setCursor(0, 1);
        lcd.print("Ya estas piso 2");
        delay(1000);
        servo1.write(93);
        lcd.print("       ");
        actual = 2;
        analogWrite(A0,0);
        analogWrite(A1,255);
        analogWrite(A2,0);        
      break;
      case 3:
        servo1.write(0);
        lcd.setCursor(0, 1);
        lcd.print("Bajando");
        delay(2000);
        servo1.write(93);
        lcd.print("       ");
        actual = 2;
        analogWrite(A0,0);
        analogWrite(A1,255);
        analogWrite(A2,0);        
      break;
    }
    delay(1000);
  }
  else if (vpul[2] == LOW) { // Ir a piso 3
    digitalWrite(13, HIGH);
    lcd.setCursor(13, 0);
    lcd.print("03");
    switch (actual) {
      case 2:
        servo1.write(180);
        lcd.setCursor(0, 1);
        lcd.print("Subiendo");
        delay(1000);
        servo1.write(93);
        lcd.print("       ");
        actual = 3;
        analogWrite(A0,0);
        analogWrite(A1,0);
        analogWrite(A2,255);
        break;
      case 1:
        servo1.write(180);
        lcd.setCursor(0, 1);
        lcd.print("Subiendo");
        delay(2000);
        servo1.write(93);
        lcd.print("       ");
        actual = 3;
        analogWrite(A0,0);
        analogWrite(A1,0);
        analogWrite(A2,255);        
      break;
      case 3:
        lcd.setCursor(0, 1);
        lcd.print("Ya estas piso 3");
        delay(1000);
        servo1.write(93);
        lcd.print("       ");
        actual = 3;
        analogWrite(A0,0);
        analogWrite(A1,0);
        analogWrite(A2,255);        
      break;
    }
    delay(1000);
  }
  else {
    digitalWrite(13, LOW);
    lcd.setCursor(0, 1);
    lcd.print("               ");
  }
}
